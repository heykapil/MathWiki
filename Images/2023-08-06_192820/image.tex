    \begin{equation*}
        \lambda
        \begin{pNiceMatrix}
            v_1 \\ \Vdots \\ \\ \\ v_n
        \end{pNiceMatrix}=
        \begin{pNiceMatrix}[nullify-dots]
            \lambda & 1      & 0      & \Cdots &        & 0      \\
            0       & \Ddots & \Ddots & \Ddots &        & \Vdots \\
            \Vdots  & \Ddots &        &        &        &        \\
                    &        &        &        &        & 0      \\
                    &        &        &        &        & 1      \\
            0       & \Cdots &        &        & 0      & \lambda
        \end{pNiceMatrix}
        \begin{pNiceMatrix}
            v_1 \\ \Vdots \\ \\ \\ v_n
        \end{pNiceMatrix}=
        \begin{pNiceMatrix}
            \lambda v_1+v_2 \\ \Vdots \\ \\ \lambda v_{n-1}+v_n \\
            \lambda v_n
        \end{pNiceMatrix}=\lambda
        \begin{pNiceMatrix}
            v_1 \\ \Vdots \\ \\ \\ v_n
        \end{pNiceMatrix}+
        \begin{pNiceMatrix}
            v_2 \\ \Vdots \\ \\ v_n \\ 0
        \end{pNiceMatrix}.
    \end{equation*}
